create table hotels ( 
	sno		integer PRIMARY KEY,
	hotelid		smallint,
	hotelname	varchar(255),
	hoteltype	varchar(50),
	hoteldesc	text,
	image		varchar(255),
	accountcode	varchar(15),
	city		varchar(50),
	paymenttype	varchar(50),
	receptionname	varchar(50),
	receptiontel	varchar(50),
	receptionfax	varchar(50),
	receptionmobile varchar(50),
	salesname	varchar(50),
	salestel	varchar(50),
	salesfax	varchar(50),
	salesmobile	varchar(50),
	reservationname	varchar(50),
	reservationtel	varchar(50),
	reservationfax	varchar(50),
	reservationmobile varchar(50),
	bankaccountname varchar(255),
	bankaccountnumber varchar(20),
	bankbranch	varchar(100),
	bankcity	varchar(100),
	notes		text
	
   );

create table rooms (
	sno		integer PRIMARY KEY,
	roomid		smallint,	
	hotelid		smallint,
	roomtype	varchar(50),
	viewtype	varchar(75),
	noofpax		smallint,
	roomdescription varchar(255),
	sdate		date,
	nrate		numeric,
	srate		numeric,
	servicecharges	numeric,
	agentcommission	numeric,
	avialibility	smallint,
	avialbool	boolean,
	breakfast	varchar(30),
	breakfastint	numeric,
	halfboard	varchar(30),
	halfboardint	numeric,
	fullboard	varchar(30),	
	fullboardint	numeric,
	sahoor		varchar(30),
	sahoorint	numeric,
	iftar		varchar(30),
	iftarint	numeric
     );
